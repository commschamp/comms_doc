267  std::size_t size,
268  TReader&& nextLayerReader,
269  TExtraValues... extraValues)
270  {

378  ErrorStatus writeInternalRandomAccess(
379  Field& field,
380  const TMsg& msg,
381  TIter& iter,
382  std::size_t size,
383  TWriter&& nextLayerWriter) const
384  {
385  auto fromIter = iter;
386  auto es = nextLayerWriter.write(msg, iter, size);
387  if ((es != comms::ErrorStatus::Success) &&
388  (es != comms::ErrorStatus::UpdateRequired)) {
389  return es;
390  }
391 
392  COMMS_ASSERT(fromIter <= iter);
393  auto len = static_cast<std::size_t>(std::distance(fromIter, iter));
394  auto remSize = size - len;

397  return comms::ErrorStatus::BufferOverflow;
398  }
399 
400  if (es == comms::ErrorStatus::UpdateRequired) {
401  auto esTmp = field.write(iter, remSize);

404  return es;
