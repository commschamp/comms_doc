203  template <typename TIter, typename TNextLayerUpdater>
204  comms::ErrorStatus doUpdate(
205  Field& field,
206  TIter& iter,
207  std::size_t size,
208  TNextLayerUpdater&& nextLayerUpdater) const
209  {
210  auto fromIter = iter;
211  auto es = nextLayerUpdater.update(iter, size - Field::maxLength());
212  if (es != comms::ErrorStatus::Success) {
213  return es;
214  }
215 
216  COMMS_ASSERT(fromIter <= iter);
217  auto len = static_cast<std::size_t>(std::distance(fromIter, iter));
218  COMMS_ASSERT(len == (size - Field::maxLength()));
219  auto remSize = size - len;
220  using FieldValueType = typename Field::ValueType;
221  field.value() = static_cast<FieldValueType>(TCalc()(fromIter, len));
222  es = field.write(iter, remSize);
223  return es;
224  }

244  template <typename TMsg, typename TIter, typename TReader, typename... TExtraValues>
245  ErrorStatus verifyRead(
246  Field& field,
247  TMsg& msg,
248  TIter& iter,
249  std::size_t size,
250  TReader&& nextLayerReader,
251  TExtraValues... extraValues)
252  {
253  auto fromIter = iter;
254  auto toIter = fromIter + (size - Field::minLength());
255  auto len = static_cast<std::size_t>(std::distance(fromIter, toIter));
256 
257  auto checksumEs = field.read(toIter, Field::minLength());
258  if (checksumEs != ErrorStatus::Success) {
259  return checksumEs;
260  }
261 
262  auto checksum = TCalc()(fromIter, len);
263  auto expectedValue = field.value();
264 
265  if (expectedValue != static_cast<decltype(expectedValue)>(checksum)) {
266  BaseImpl::resetMsg(msg);
267  return ErrorStatus::ProtocolError;
268  }
269 
270  auto es = nextLayerReader.read(msg, iter, size - Field::minLength(), extraValues...);
271  if (es == ErrorStatus::Success) {
272  iter = toIter;
273  }
274 
275  return es;
276  }

279  ErrorStatus readVerify(
280  Field& field,
281  TMsg& msg,
282  TIter& iter,
283  std::size_t size,
284  TReader&& nextLayerReader,
285  TExtraValues... extraValues)
286  {
287  auto fromIter = iter;
288 
289  auto es = nextLayerReader.read(msg, iter, size - Field::minLength(), extraValues...);
290  if ((es == ErrorStatus::NotEnoughData) ||
291  (es == ErrorStatus::ProtocolError)) {
292  return es;
293  }
294

296  COMMS_ASSERT(len <= size);
297  auto remSize = size - len;
298  auto checksumEs = field.read(iter, remSize);
299  if (checksumEs == ErrorStatus::NotEnoughData) {

302

304  BaseImpl::resetMsg(msg);
305  return checksumEs;
306  }
307 
308  auto checksum = TCalc()(fromIter, len);
309  auto expectedValue = field.value();
310 
311  if (expectedValue != static_cast<decltype(expectedValue)>(checksum)) {
312  BaseImpl::resetMsg(msg);
313  return ErrorStatus::ProtocolError;
314  }

316  return es;

359  template <typename TMsg, typename TIter, typename TWriter>
360  ErrorStatus writeInternalRandomAccess(
361  Field& field,
362  const TMsg& msg,
363  TIter& iter,
364  std::size_t size,
365  TWriter&& nextLayerWriter) const
366  {
367  auto fromIter = iter;
368  auto es = nextLayerWriter.write(msg, iter, size);
369  if ((es != comms::ErrorStatus::Success) &&
370  (es != comms::ErrorStatus::UpdateRequired)) {
371  return es;
372  }
373 
374  COMMS_ASSERT(fromIter <= iter);
375  auto len = static_cast<std::size_t>(std::distance(fromIter, iter));
376  auto remSize = size - len;
377 
378  if (remSize < Field::maxLength()) {
379  return comms::ErrorStatus::BufferOverflow;
380  }
381 
382  if (es == comms::ErrorStatus::UpdateRequired) {
383  auto esTmp = field.write(iter, remSize);
384  static_cast<void>(esTmp);
385  COMMS_ASSERT(esTmp == comms::ErrorStatus::Success);
386  return es;
387  }
388 
389  using FieldValueType = typename Field::ValueType;
390  auto checksum = TCalc()(fromIter, len);
391  field.value() = static_cast<FieldValueType>(checksum);
392 
393  return field.write(iter, remSize);
394  }
