376  ErrorStatus writeInternalRandomAccess(
377  Field& field,
378  const TMsg& msg,
379  TIter& iter,
380  std::size_t size,
381  TWriter&& nextLayerWriter) const
382  {
383  auto checksumIter = iter;

385  if (es != comms::ErrorStatus::Success) {
386  return es;

389  auto checksumLen =

391 
392  auto fromIter = iter;
393  es = nextLayerWriter.write(msg, iter, size - checksumLen);
394  if (es != comms::ErrorStatus::Success) {
395  return es;
396  }
397 
398  COMMS_ASSERT(fromIter <= iter);
399  auto len = static_cast<std::size_t>(std::distance(fromIter, iter));
400 
401  using FieldValueType = typename Field::ValueType;
402  auto checksum = TCalc()(fromIter, len);
403  field.value() = static_cast<FieldValueType>(checksum);
404 
405  auto checksumEs = field.write(checksumIter, checksumLen);
406  static_cast<void>(checksumEs);
407  COMMS_ASSERT(checksumEs == comms::ErrorStatus::Success);
408  return es;
