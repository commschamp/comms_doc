32 template <typename TResult>
33 using CrcInitTableType = std::array<TResult, 256>;

35 template <typename TResult, TResult TPoly>
36 struct CrcInitTable
37 {
38  using Table = CrcInitTableType<TResult>;
39  static const Table& get()
40  {
41  static Table table;
42  static bool tableFilled = false;
43 
44  if (!tableFilled) {
45  fillTable(table);
46  tableFilled = true;
47  }
48 
49  return table;
50  }
51 
52 private:
53  static void fillTable(Table& table)
54  {
55  static const std::size_t Width =
56  sizeof(TResult) * std::numeric_limits<std::uint8_t>::digits;
57  static const auto Msb =
58  static_cast<TResult>(1) << (Width - 1);
59 
60  for (unsigned idx = 0U; idx < table.size(); ++idx)
61  {
62  auto rem = static_cast<TResult>(idx << (Width - 8));
63 
64  for (auto bit = 8U; bit > 0U; --bit)
65  {
66  if ((rem & Msb) != 0)
67  {
68  rem = (rem << 1) ^ TPoly;
69  }
70  else
71  {
72  rem = (rem << 1);
73  }
74  }
75 
76  table[idx] = rem;
77  }
78  }
79 };

258 template <
259  typename TResult,
260  TResult TPoly,
261  TResult TInit = 0,
262  TResult TFin = 0,
263  bool TReflect = false,
264  bool TRefrectRem = false
265 >
266 class Crc
267 {
268  static_assert(std::is_unsigned<TResult>::value,
269  "The TResult type is expected to be unsigned integral one");
270 public:
276  template <typename TIter>
277  TResult operator()(TIter& iter, std::size_t len) const
278  {
279  static const std::size_t Width =
280  sizeof(TResult) * std::numeric_limits<std::uint8_t>::digits;
281 
282  TResult rem = TInit;
283  auto& initTable = details::CrcInitTable<TResult, TPoly>::get();
284 
285  for (std::size_t byte = 0U; byte < len; ++byte)
286  {
287  using ByteType = typename std::make_unsigned<
288  typename std::decay<decltype(*iter)>::type
289  >::type;
290 
291  auto val = static_cast<std::uint8_t>(static_cast<ByteType>(*iter));
292  val = reflect(val) ^ static_cast<decltype(val)>(rem >> (Width - 8));
293  rem = initTable[val] ^ static_cast<decltype(rem)>(rem << 8);
294  ++iter;
295  }
296 
297  return (reflectRem(rem) ^ TFin);
298  }
299 
300 private:
301  template <typename... TParams>
302  using NoReflectTag = comms::details::tag::Tag1<>;
303 
304  template <typename... TParams>
305  using DoReflectTag = comms::details::tag::Tag2<>;
306 
307  template <typename...>
308  using ReflectTag =
309  typename comms::util::LazyShallowConditional<
310  TReflect
311  >::template Type<
312  DoReflectTag,
313  NoReflectTag
314  >;
315 
316  template <typename...>
317  using ReflectRemTag =
318  typename comms::util::LazyShallowConditional<
319  TRefrectRem
320  >::template Type<
321  DoReflectTag,
322  NoReflectTag
323  >;
324 
325  static std::uint8_t reflect(std::uint8_t byte)
326  {
327  return reflectInternal(byte, 8U, ReflectTag<>());
328  }
329 
330  static TResult reflectRem(TResult value)
331  {
332  static const std::size_t Width =
333  sizeof(TResult) * std::numeric_limits<std::uint8_t>::digits;
334 
335  return reflectInternal(value, Width, ReflectRemTag<>());
336  }
337 
338  template <typename TVal, typename... TParams>
339  static TVal reflectInternal(TVal value, std::size_t bitsCount, DoReflectTag<TParams...>)
340  {
341  return static_cast<TVal>(doReflect(value, bitsCount));
342  }
343 
344  template <typename TVal, typename... TParams>
345  static constexpr TVal reflectInternal(TVal value, std::size_t, NoReflectTag<TParams...>)
346  {
347  return value;
348  }
349 
350  static TResult doReflect(TResult value, std::size_t bitsCount)
351  {
352  TResult reflection = 0U;
353  for (auto bit = 0U; bit < bitsCount; ++bit)
354  {
355  if (value & 0x01)
356  {
357  reflection |= static_cast<decltype(reflection)>(1 << ((bitsCount - 1) - bit));
358  }
359 
360  value = static_cast<decltype(value)>(value >> 1);
361  }
362 
363  return (reflection);
364  }
