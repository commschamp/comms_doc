359  template <typename TScaled>
360  void setScaledInternal(TScaled val, ScaleAsFpTag)
361  {
362  using DecayedType = typename std::decay<decltype(val)>::type;
363  auto epsilon = DecayedType(0);
364  if (ParsedOptions::ScalingRatio::num < ParsedOptions::ScalingRatio::den) {
365  epsilon = static_cast<DecayedType>(ParsedOptions::ScalingRatio::num) / static_cast<DecayedType>(ParsedOptions::ScalingRatio::den + 1);
366  }
367 
368  if (epsilon < DecayedType(0)) {
369  epsilon = -epsilon;
370  }
371 
372  if (val < DecayedType(0)) {
373  epsilon = -epsilon;
374  }
375 
376  BaseImpl::value() =
377  static_cast<ValueType>(
378  ((val + epsilon) * static_cast<DecayedType>(ParsedOptions::ScalingRatio::den)) / static_cast<DecayedType>(ParsedOptions::ScalingRatio::num));
379  }
