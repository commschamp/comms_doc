368  template <typename TScaled, typename... TParams>
369  void setScaledInternal(TScaled val, ScaleAsFpTag<TParams...>)
370  {
371  using DecayedType = typename std::decay<decltype(val)>::type;
372  auto epsilon = DecayedType(0);
373  if (ParsedOptions::ScalingRatio::num < ParsedOptions::ScalingRatio::den) {
374  epsilon = static_cast<DecayedType>(ParsedOptions::ScalingRatio::num) / static_cast<DecayedType>(ParsedOptions::ScalingRatio::den + 1);
375  }
376 
377  if (epsilon < DecayedType(0)) {
378  epsilon = -epsilon;
379  }
380 
381  if (val < DecayedType(0)) {
382  epsilon = -epsilon;
383  }
384 
385  BaseImpl::value() =
386  static_cast<ValueType>(
387  ((val + epsilon) * static_cast<DecayedType>(ParsedOptions::ScalingRatio::den)) / static_cast<DecayedType>(ParsedOptions::ScalingRatio::num));
388  }
