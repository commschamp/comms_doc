637  template <typename TMsg, typename TIter, typename... TExtraValues>
638  static ErrorStatus readInternalPolymorphic(
639  TMsg& msg,
640  TIter& iter,
641  std::size_t size,
642  TExtraValues... extraValues)
643  {
644  using MsgType = typename std::decay<decltype(msg)>::type;
645 
646  static_assert(
647  comms::isMessage<MsgType>(),
648  "The provided message object must inherit from comms::Message");
649 
650  static_assert(MsgType::hasRead(),
651  "Message interface must support polymorphic read operation");
652 
653  using IterType = typename std::decay<decltype(iter)>::type;
654 
655  static_assert(std::is_convertible<IterType, typename MsgType::ReadIterator>::value,
656  "The provided iterator is not convertible to ReadIterator defined by Message class");
657 
658  using ReadIter = typename std::add_lvalue_reference<typename MsgType::ReadIterator>::type;
659 
660  auto result = msg.read(static_cast<ReadIter>(iter), size);
661  if ((result == ErrorStatus::NotEnoughData) &&
662  missingSizeRequiredInternal(extraValues...)) {
663  using Tag =
664  typename comms::util::LazyShallowConditional<
665  MsgType::InterfaceOptions::HasLength
666  >::template Type<
667  MsgHasLengthTag,
668  MsgNoLengthTag
669  >;
670 
671  std::size_t missingSize = 1U;
672  auto msgLen = getMsgLength(msg, Tag());
673  if (size < msgLen) {
674  missingSize = msgLen - size;
675  }
676  updateMissingSizeInternal(missingSize, extraValues...);
677  }
678  return result;
679  }
