132  MsgPtr createMsg(MsgIdParamType id, unsigned idx, CreateFailureReason* reason) const
133  {
134  CreateFailureReason reasonTmp = CreateFailureReason::None;
135  bool result = false;
136  MsgPtr msg = createMsgInternal(id, idx, result, DestructorTag<>());
137  do {
138  if (msg) {
139  COMMS_ASSERT(result);
140  break;
141  }
142 
143  if (!result) {
144  reasonTmp = CreateFailureReason::InvalidId;
145  break;
146  }
147 
148  reasonTmp = CreateFailureReason::AllocFailure;
149  } while (false);
150 
151  if (reason != nullptr) {
152  *reason = reasonTmp;
153  }
154 
155  return msg;
156  }
