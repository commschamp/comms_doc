154  template <typename TMsg, typename TIter, typename TNextLayerReader, typename... TExtraValues>
155  comms::ErrorStatus doRead(
156  Field& field,
157  TMsg& msg,
158  TIter& iter,
159  std::size_t size,
160  TNextLayerReader&& nextLayerReader,
161  TExtraValues... extraValues)
162  {
163  using IterType = typename std::decay<decltype(iter)>::type;
164  using IterTag = typename std::iterator_traits<IterType>::iterator_category;
165  static_assert(
166  std::is_base_of<std::random_access_iterator_tag, IterTag>::value,
167  "Current implementation of MsgSizeLayer requires iterator used for reading to be random-access one.");
168 
169  auto begIter = iter;
170  auto* msgPtr = BaseImpl::toMsgPtr(msg);
171  auto& thisObj = BaseImpl::thisLayer();
172  auto es = thisObj.doReadField(msgPtr, field, iter, size);
173  if (es == ErrorStatus::NotEnoughData) {
174  BaseImpl::updateMissingSize(field, size, extraValues...);
175  }
176 
177  if (es != ErrorStatus::Success) {
178  return es;
179  }
180 
181  auto fromIter = iter;
182  auto readFieldLength = static_cast<std::size_t>(std::distance(begIter, iter));
183  std::size_t actualRemainingSize = (size - readFieldLength);
184  std::size_t requiredRemainingSize = thisObj.getRemainingSizeFromField(field);
185 
186  if (actualRemainingSize < requiredRemainingSize) {
187  BaseImpl::setMissingSize(requiredRemainingSize - actualRemainingSize, extraValues...);
188  return ErrorStatus::NotEnoughData;
189  }
190 
191  thisObj.beforeRead(field, msgPtr);
192  es = nextLayerReader.read(msg, iter, requiredRemainingSize, extraValues...);
193  if (es == ErrorStatus::NotEnoughData) {
194  BaseImpl::resetMsg(msg);
195  return ErrorStatus::ProtocolError;
196  }
197 
198  if (es != ErrorStatus::ProtocolError) {
199  iter = fromIter;
200  std::advance(iter, requiredRemainingSize);
201  }
202 
203  auto consumed =
204  static_cast<std::size_t>(std::distance(fromIter, iter));
205  if (consumed < requiredRemainingSize) {
206  auto diff = requiredRemainingSize - consumed;
207  std::advance(iter, diff);
208  }
209  return es;
210  }

516  template <typename TMsg, typename TIter, typename TNextLayerUpdater, typename TForwardTag>
517  comms::ErrorStatus doUpdateInternal(
518  const TMsg* msg,
519  Field& field,
520  TIter& iter,
521  std::size_t size,
522  TNextLayerUpdater&& nextLayerUpdater,
523  TForwardTag&& tag) const
524  {
525  std::size_t lenValue = size - Field::maxLength();
526  auto& thisObj = BaseImpl::thisLayer();
527  thisObj.prepareFieldForWrite(lenValue, msg, field);
528

531  thisObj.prepareFieldForWrite(lenValue, msg, field);
532  }
533

535  if (es != ErrorStatus::Success) {
536  return es;

539  return doUpdateForward(msg, iter, size - field.length(), std::forward<TNextLayerUpdater>(nextLayerUpdater), tag);
