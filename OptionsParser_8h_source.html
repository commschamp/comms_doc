569 template <bool THasMultiRangeValidation>
570 struct MultiRangeAssembler
571 {
572  template <typename TBase, typename T, T TMinValue, T TMaxValue>
573  using Type =
574  std::tuple<
575  std::tuple<
576  std::integral_constant<T, TMinValue>,
577  std::integral_constant<T, TMaxValue>
578  >
579  >;
580 };
581 
582 template <>
583 struct MultiRangeAssembler<true>
584 {
585  using FalseAssembler = MultiRangeAssembler<false>;
586 
587  template <typename TBase, typename T, T TMinValue, T TMaxValue>
588  using Type =
589  typename std::decay<
590  decltype(
591  std::tuple_cat(
592  std::declval<typename TBase::MultiRangeValidationRanges>(),
593  std::declval<typename FalseAssembler::template Type<TBase, T, TMinValue, TMaxValue> >()
594  )
595  )
596  >::type;
597 };
598
