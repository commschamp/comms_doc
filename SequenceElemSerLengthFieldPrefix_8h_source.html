101  template <typename TIter>
102  ErrorStatus readElement(ElementType& elem, TIter& iter, std::size_t& len) const
103  {
104  auto fromIter = iter;
105  LenField lenField;
106  auto es = lenField.read(iter, len);
107  if (es != ErrorStatus::Success) {
108  return es;
109  }
110 
111  auto diff = static_cast<std::size_t>(std::distance(fromIter, iter));
112  COMMS_ASSERT(diff <= len);
113  len -= diff;
114  if (len < lenField.value()) {
115  return comms::ErrorStatus::NotEnoughData;
116  }
117 
118  const auto reqLen = static_cast<std::size_t>(lenField.value());
119  std::size_t elemLen = reqLen;
120  es = BaseImpl::readElement(elem, iter, elemLen);
121  if (es == ErrorStatus::NotEnoughData) {
122  return TStatus;
123  }
124 
125  if (es != ErrorStatus::Success) {
126  return es;
127  }
128 
129  COMMS_ASSERT(elemLen <= reqLen);
130  std::advance(iter, elemLen);
131  len -= reqLen;
132  return ErrorStatus::Success;
133  }
