208  template <typename U>
209  T* insert(const T* pos, U&& value)
210  {
211  COMMS_ASSERT(pos <= end());
212  COMMS_ASSERT(size() < capacity());
213  if (end() <= pos) {
214  push_back(std::forward<U>(value));
215  return &(back());
216  }
217 
218  COMMS_ASSERT(!empty());
219  push_back(std::move(back()));
220  auto* insertIter = begin() + std::distance(cbegin(), pos);
221  std::move_backward(insertIter, end() - 2, end() - 1);
222  *insertIter = std::forward<U>(value);
223  return insertIter;
224  }

226  T* insert(const T* pos, std::size_t count, const T& value)
227  {
228  COMMS_ASSERT(pos <= end());
229  COMMS_ASSERT((size() + count) <= capacity());
230  auto* posIter = begin() + std::distance(cbegin(), pos);
231  if (end() <= posIter) {
232  while (0 < count) {
233  push_back(value);
234  --count;
235  }
236  return posIter;
237  }
238 
239  COMMS_ASSERT(!empty());
240  auto tailCount = static_cast<std::size_t>(std::distance(posIter, end()));
241  if (count <= tailCount) {
242  auto pushBegIter = end() - count;
243  auto pushEndIter = end();
244  for (auto iter = pushBegIter; iter != pushEndIter; ++iter) {
245  push_back(std::move(*iter));
246  }
247 
248  auto moveBegIter = posIter;
249  auto moveEndIter = moveBegIter + (tailCount - count);
250  COMMS_ASSERT(moveEndIter < pushEndIter);
251  std::move_backward(moveBegIter, moveEndIter, pushEndIter);
252 
253  auto* assignBegIter = posIter;
254  auto* assignEndIter = assignBegIter + count;
255  for (auto iter = assignBegIter; iter != assignEndIter; ++iter) {
256  *iter = value;
257  }
258  return posIter;
259  }
260 
261  auto pushValueCount = count - tailCount;
262  for (auto idx = 0U; idx < pushValueCount; ++idx) {
263  push_back(value);
264  }
265 
266  auto* pushBegIter = posIter;
267  auto* pushEndIter = pushBegIter + tailCount;
268  for (auto iter = pushBegIter; iter != pushEndIter; ++iter) {
269  push_back(std::move(*iter));
270  }
271 
272  auto assignBegIter = posIter;
273  auto assignEndIter = assignBegIter + tailCount;
274  for (auto iter = assignBegIter; iter != assignEndIter; ++iter) {
275  *iter = value;
276  }
277  return posIter;
278  }

287  template <typename... TArgs>
288  T* emplace(const T* iter, TArgs&&... args)
289  {
290  auto* insertIter = begin() + std::distance(cbegin(), iter);
291  if (iter == cend()) {
292  emplace_back(std::forward<TArgs>(args)...);
293  return insertIter;
294  }
295 
296  COMMS_ASSERT(!empty());
297  push_back(std::move(back()));
298  std::move_backward(insertIter, end() - 2, end() - 1);
299  insertIter->~T();
300  new (insertIter) T(std::forward<TArgs>(args)...);
301  return insertIter;

304  T* erase(const T* from, const T* to)
305  {
306  COMMS_ASSERT(from <= cend());
307  COMMS_ASSERT(to <= cend());
308  COMMS_ASSERT(from <= to);
309 
310  auto tailCount = static_cast<std::size_t>(std::distance(to, cend()));
311  auto eraseCount = static_cast<std::size_t>(std::distance(from, to));
312 
313  auto* moveSrc = begin() + std::distance(cbegin(), to);
314  auto* moveDest = begin() + std::distance(cbegin(), from);
315  std::move(moveSrc, end(), moveDest);
316 
317  auto* eraseFrom = moveDest + tailCount;
318  auto* eraseTo = end();
319  COMMS_ASSERT(eraseFrom <= end());
320  COMMS_ASSERT(eraseCount <= size());
321  COMMS_ASSERT(static_cast<std::size_t>(std::distance(eraseFrom, eraseTo)) == eraseCount);
322  for (auto iter = eraseFrom; iter != eraseTo; ++iter) {
323  iter->~T();
324  }
325  size_ -= eraseCount;
326  return moveDest;
327  }

358  void swap(StaticVectorBase<T>& other)
359  {
360  auto swapSize = std::min(other.size(), size());
361  for (auto idx = 0U; idx < swapSize; ++idx) {
362  std::swap(this->operator[](idx), other[idx]);
363  }
364 
365  auto otherSize = other.size();
366  auto thisSize = size();
367 
368  if (otherSize == thisSize) {
369  return;
370  }
371 
372  if (otherSize < thisSize) {
373  auto limit = std::min(thisSize, other.capacity());
374  for (auto idx = swapSize; idx < limit; ++idx) {
375  new (other.cellPtr(idx)) T(std::move(elem(idx)));
376  }
377 
378  other.size_ = thisSize;
379  erase(begin() + otherSize, end());
380  return;
381  }
382 
383  auto limit = std::min(otherSize, capacity());
384  for (auto idx = swapSize; idx < limit; ++idx) {
385  new (cellPtr(idx)) T(std::move(other.elem(idx)));
386  }
387  size_ = otherSize;
388  other.erase(other.begin() + thisSize, other.end());
389  }

420  template <typename TIter>
421  T* insert_random_access(const T* pos, TIter from, TIter to)
422  {
423  COMMS_ASSERT(pos <= end());
424  auto* posIter = begin() + std::distance(cbegin(), pos);
425  if (end() <= posIter) {
426  for (; from != to; ++from) {
427  push_back(*from);
428  }
429 
430  return posIter;
431  }
432 
433  auto count = static_cast<std::size_t>(std::distance(from, to));
434  COMMS_ASSERT(!empty());
435  auto tailCount = static_cast<std::size_t>(std::distance(posIter, end()));
436  if (count <= tailCount) {
437  auto pushBegIter = end() - count;
438  auto pushEndIter = end();
439  for (auto iter = pushBegIter; iter != pushEndIter; ++iter) {
440  push_back(std::move(*iter));
441  }
442 
443  auto moveBegIter = posIter;
444  auto moveEndIter = moveBegIter + (tailCount - count);
445  COMMS_ASSERT(moveEndIter < pushEndIter);
446  std::move_backward(moveBegIter, moveEndIter, pushEndIter);
447 
448  auto* assignBegIter = posIter;
449  auto* assignEndIter = assignBegIter + count;
450  for (auto iter = assignBegIter; iter != assignEndIter; ++iter) {
451  *iter = *from;
452  ++from;
453  }
454  return posIter;
455  }
456 
457  auto pushValueCount = count - tailCount;
458  auto pushInsertedBegIter = to - pushValueCount;
459  for (auto idx = 0U; idx < pushValueCount; ++idx) {
460  push_back(*pushInsertedBegIter);
461  ++pushInsertedBegIter;
462  }
463 
464  auto* pushBegIter = posIter;
465  auto* pushEndIter = pushBegIter + tailCount;
466  for (auto iter = pushBegIter; iter != pushEndIter; ++iter) {
467  push_back(std::move(*iter));
468  }
469 
470  auto assignBegIter = posIter;
471  auto assignEndIter = assignBegIter + tailCount;
472  for (auto iter = assignBegIter; iter != assignEndIter; ++iter) {
473  *iter = *from;
474  ++from;
475  }
476 
477  return posIter;
478  }

481  T* insert_input(const T* pos, TIter from, TIter to)
