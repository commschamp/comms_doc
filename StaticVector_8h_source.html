206  template <typename U>
207  T* insert(const T* pos, U&& value)
208  {
209  COMMS_ASSERT(pos <= end());
210  COMMS_ASSERT(size() < capacity());
211  if (end() <= pos) {
212  push_back(std::forward<U>(value));
213  return &(back());
214  }
215 
216  COMMS_ASSERT(!empty());
217  push_back(std::move(back()));
218  auto* insertIter = begin() + std::distance(cbegin(), pos);
219  std::move_backward(insertIter, end() - 2, end() - 1);
220  *insertIter = std::forward<U>(value);
221  return insertIter;
222  }

224  T* insert(const T* pos, std::size_t count, const T& value)
225  {
226  COMMS_ASSERT(pos <= end());
227  COMMS_ASSERT((size() + count) <= capacity());
228  auto* posIter = begin() + std::distance(cbegin(), pos);
229  if (end() <= posIter) {
230  while (0 < count) {
231  push_back(value);
232  --count;
233  }
234  return posIter;
235  }
236 
237  COMMS_ASSERT(!empty());
238  auto tailCount = static_cast<std::size_t>(std::distance(posIter, end()));
239  if (count <= tailCount) {
240  auto pushBegIter = end() - count;
241  auto pushEndIter = end();
242  for (auto iter = pushBegIter; iter != pushEndIter; ++iter) {
243  push_back(std::move(*iter));
244  }
245 
246  auto moveBegIter = posIter;
247  auto moveEndIter = moveBegIter + (tailCount - count);
248  COMMS_ASSERT(moveEndIter < pushEndIter);
249  std::move_backward(moveBegIter, moveEndIter, pushEndIter);
250 
251  auto* assignBegIter = posIter;
252  auto* assignEndIter = assignBegIter + count;
253  for (auto iter = assignBegIter; iter != assignEndIter; ++iter) {
254  *iter = value;
255  }
256  return posIter;
257  }
258 
259  auto pushValueCount = count - tailCount;
260  for (auto idx = 0U; idx < pushValueCount; ++idx) {
261  push_back(value);
262  }
263 
264  auto* pushBegIter = posIter;
265  auto* pushEndIter = pushBegIter + tailCount;
266  for (auto iter = pushBegIter; iter != pushEndIter; ++iter) {
267  push_back(std::move(*iter));
268  }
269 
270  auto assignBegIter = posIter;
271  auto assignEndIter = assignBegIter + tailCount;
272  for (auto iter = assignBegIter; iter != assignEndIter; ++iter) {
273  *iter = value;
274  }
275  return posIter;
276  }

285  template <typename... TArgs>
286  T* emplace(const T* iter, TArgs&&... args)
287  {
288  auto* insertIter = begin() + std::distance(cbegin(), iter);
289  if (iter == cend()) {
290  emplace_back(std::forward<TArgs>(args)...);
291  return insertIter;
292  }

295  push_back(std::move(back()));
296  std::move_backward(insertIter, end() - 2, end() - 1);
297  insertIter->~T();
298  new (insertIter) T(std::forward<TArgs>(args)...);

302  T* erase(const T* from, const T* to)
303  {
304  COMMS_ASSERT(from <= cend());
305  COMMS_ASSERT(to <= cend());
306  COMMS_ASSERT(from <= to);
307 
308  auto tailCount = static_cast<std::size_t>(std::distance(to, cend()));
309  auto eraseCount = static_cast<std::size_t>(std::distance(from, to));
310 
311  auto* moveSrc = begin() + std::distance(cbegin(), to);
312  auto* moveDest = begin() + std::distance(cbegin(), from);
313  std::move(moveSrc, end(), moveDest);
314 
315  auto* eraseFrom = moveDest + tailCount;
316  auto* eraseTo = end();
317  COMMS_ASSERT(eraseFrom <= end());
318  COMMS_ASSERT(eraseCount <= size());
319  COMMS_ASSERT(static_cast<std::size_t>(std::distance(eraseFrom, eraseTo)) == eraseCount);
320  for (auto iter = eraseFrom; iter != eraseTo; ++iter) {
321  iter->~T();
322  }
323  size_ -= eraseCount;
324  return moveDest;
325  }

356  void swap(StaticVectorBase<T>& other)
357  {
358  auto swapSize = std::min(other.size(), size());
359  for (auto idx = 0U; idx < swapSize; ++idx) {
360  std::swap(this->operator[](idx), other[idx]);
361  }
362 
363  auto otherSize = other.size();
364  auto thisSize = size();
365 
366  if (otherSize == thisSize) {
367  return;
368  }
369 
370  if (otherSize < thisSize) {
371  auto limit = std::min(thisSize, other.capacity());
372  for (auto idx = swapSize; idx < limit; ++idx) {
373  new (other.cellPtr(idx)) T(std::move(elem(idx)));
374  }
375 
376  other.size_ = thisSize;
377  erase(begin() + otherSize, end());
378  return;
379  }
380 
381  auto limit = std::min(otherSize, capacity());
382  for (auto idx = swapSize; idx < limit; ++idx) {
383  new (cellPtr(idx)) T(std::move(other.elem(idx)));
384  }
385  size_ = otherSize;
386  other.erase(other.begin() + thisSize, other.end());
387  }

418  template <typename TIter>
419  T* insert_random_access(const T* pos, TIter from, TIter to)
420  {
421  COMMS_ASSERT(pos <= end());
422  auto* posIter = begin() + std::distance(cbegin(), pos);
423  if (end() <= posIter) {
424  for (; from != to; ++from) {
425  push_back(*from);
426  }
427 
428  return posIter;
429  }
430 
431  auto count = static_cast<std::size_t>(std::distance(from, to));
432  COMMS_ASSERT(!empty());
433  auto tailCount = static_cast<std::size_t>(std::distance(posIter, end()));
434  if (count <= tailCount) {
435  auto pushBegIter = end() - count;
436  auto pushEndIter = end();
437  for (auto iter = pushBegIter; iter != pushEndIter; ++iter) {
438  push_back(std::move(*iter));
439  }
440 
441  auto moveBegIter = posIter;
442  auto moveEndIter = moveBegIter + (tailCount - count);
443  COMMS_ASSERT(moveEndIter < pushEndIter);
444  std::move_backward(moveBegIter, moveEndIter, pushEndIter);
445 
446  auto* assignBegIter = posIter;
447  auto* assignEndIter = assignBegIter + count;
448  for (auto iter = assignBegIter; iter != assignEndIter; ++iter) {
449  *iter = *from;
450  ++from;
451  }
452  return posIter;
453  }
454 
455  auto pushValueCount = count - tailCount;
456  auto pushInsertedBegIter = to - pushValueCount;
457  for (auto idx = 0U; idx < pushValueCount; ++idx) {
458  push_back(*pushInsertedBegIter);
459  ++pushInsertedBegIter;
460  }
461 
462  auto* pushBegIter = posIter;
463  auto* pushEndIter = pushBegIter + tailCount;
464  for (auto iter = pushBegIter; iter != pushEndIter; ++iter) {
465  push_back(std::move(*iter));
466  }
467 
468  auto assignBegIter = posIter;
469  auto assignEndIter = assignBegIter + tailCount;
470  for (auto iter = assignBegIter; iter != assignEndIter; ++iter) {
471  *iter = *from;
472  ++from;
473  }
474 
475  return posIter;
476  }
