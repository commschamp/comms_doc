107  template <typename TMsg, typename TIter, typename TNextLayerReader, typename... TExtraValues>
108  comms::ErrorStatus doRead(
109  Field& field,
110  TMsg& msg,
111  TIter& iter,
112  std::size_t size,
113  TNextLayerReader&& nextLayerReader,
114  TExtraValues... extraValues)
115  {
116  auto& thisObj = BaseImpl::thisLayer();
117  auto* msgPtr = BaseImpl::toMsgPtr(msg);
118  auto beforeReadIter = iter;
119 
120  auto es = thisObj.doReadField(msgPtr, field, iter, size);
121  if (es == comms::ErrorStatus::NotEnoughData) {
122  BaseImpl::updateMissingSize(field, size, extraValues...);
123  }
124 
125  if (es != comms::ErrorStatus::Success) {
126  return es;
127  }
128 
129  bool verified = thisObj.verifyFieldValue(field);
130  if (!verified) {
131  return comms::ErrorStatus::ProtocolError;
132  }
133 
134  auto fieldLen = static_cast<std::size_t>(std::distance(beforeReadIter, iter));
135  return nextLayerReader.read(msg, iter, size - fieldLen, extraValues...);
136  }

155  template <typename TMsg, typename TIter, typename TNextLayerWriter>
156  comms::ErrorStatus doWrite(
157  Field& field,
158  const TMsg& msg,
159  TIter& iter,
160  std::size_t size,
161  TNextLayerWriter&& nextLayerWriter) const
162  {
163  auto& thisObj = BaseImpl::thisLayer();
164  thisObj.prepareFieldForWrite(field);
165  auto es = thisObj.doWriteField(&msg, field, iter, size);
166  if (es != ErrorStatus::Success) {
167  return es;
168  }
169 
170  COMMS_ASSERT(field.length() <= size);
171  return nextLayerWriter.write(msg, iter, size - field.length());
