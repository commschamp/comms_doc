171  comms::ErrorStatus doWrite(

176  TNextLayerWriter&& nextLayerWriter) const
177  {
178  using MsgType = typename std::decay<decltype(msg)>::type;
179  static_assert(MsgType::hasTransportFields(),
180  "Message interface class hasn't defined transport fields, "
181  "use comms::option::def::ExtraTransportFields option.");
182  static_assert(TIdx < std::tuple_size<typename MsgType::TransportFields>::value,
183  "TIdx is too big, exceeds the amount of transport fields defined in interface class");
184 
185  auto& transportField = std::get<TIdx>(msg.transportFields());
186  field = comms::field_cast<Field>(transportField);
187 
188  auto es = writeFieldInternal(field, iter, size, ValueTag());
189  if (es != ErrorStatus::Success) {
190  return es;
191  }
192 
193  return nextLayerWriter.write(msg, iter, size);

301  template <typename TIter, typename... TExtraValues>
302  comms::ErrorStatus readFieldInternal(
303  Field& field,
304  TIter& iter,
305  std::size_t& len,
306  NormalValueTag,
307  TExtraValues... extraValues)
308  {
309  auto beforeReadIter = iter;
310  auto es = field.read(iter, len);
311  if (es == comms::ErrorStatus::NotEnoughData) {
312  BaseImpl::updateMissingSize(field, len, extraValues...);
313  }
314  else {
315  auto fieldLen = static_cast<std::size_t>(std::distance(beforeReadIter, iter));
316  len -= fieldLen;
317  }
318  return es;
319  }
