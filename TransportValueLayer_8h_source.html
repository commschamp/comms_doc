174  comms::ErrorStatus doWrite(
175  Field& field,

178  std::size_t size,
179  TNextLayerWriter&& nextLayerWriter) const
180  {
181  using MsgType = typename std::decay<decltype(msg)>::type;
182  static_assert(MsgType::hasTransportFields(),
183  "Message interface class hasn't defined transport fields, "
184  "use comms::option::def::ExtraTransportFields option.");
185  static_assert(TIdx < std::tuple_size<typename MsgType::TransportFields>::value,
186  "TIdx is too big, exceeds the amount of transport fields defined in interface class");
187 
188  auto& transportField = std::get<TIdx>(msg.transportFields());
189  field = comms::field_cast<Field>(transportField);
190 
191  auto es = writeFieldInternal(field, iter, size, ValueTag<>());
192  if (es != ErrorStatus::Success) {
193  return es;
194  }
195

315  template <typename TIter, typename... TExtraValues>
316  comms::ErrorStatus readFieldInternal(
317  Field& field,
318  TIter& iter,
319  std::size_t& len,
320  NormalValueTag<>,
321  TExtraValues... extraValues)
322  {
323  auto beforeReadIter = iter;
324  auto es = field.read(iter, len);
325  if (es == comms::ErrorStatus::NotEnoughData) {
326  BaseImpl::updateMissingSize(field, len, extraValues...);
327  }
328  else {
329  auto fieldLen = static_cast<std::size_t>(std::distance(beforeReadIter, iter));
330  len -= fieldLen;
331  }
332  return es;
333  }
