503 template <typename TFieldBase, typename TMembers, typename... TOptions>
504 bool operator==(
505  const Variant<TFieldBase, TMembers, TOptions...>& field1,
506  const Variant<TFieldBase, TMembers, TOptions...>& field2)
507 {
508  if (&field1 == &field2) {
509  return true;
510  }
511 
512  if (field1.currentFieldValid() != field2.currentFieldValid()) {
513  return false;
514  }
515 
516  if (!field1.currentFieldValid()) {
517  return true;
518  }
519 
520  if (field1.currentField() != field2.currentField()) {
521  return false;
522  }
523 
524  bool result = false;
525  field1.currentFieldExec(details::makeVariantEqualityCompHelper(field2, result));
526  return result;
527 }
