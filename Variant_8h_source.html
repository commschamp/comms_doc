512 template <typename TFieldBase, typename TMembers, typename... TOptions>
513 bool operator==(
514  const Variant<TFieldBase, TMembers, TOptions...>& field1,
515  const Variant<TFieldBase, TMembers, TOptions...>& field2)
516 {
517  if (&field1 == &field2) {
518  return true;
519  }
520 
521  if (field1.currentFieldValid() != field2.currentFieldValid()) {
522  return false;
523  }
524 
525  if (!field1.currentFieldValid()) {
526  return true;
527  }
528 
529  if (field1.currentField() != field2.currentField()) {
530  return false;
531  }
532 
533  bool result = false;
534  field1.currentFieldExec(details::makeVariantEqualityCompHelper(field2, result));
535  return result;
536 }
