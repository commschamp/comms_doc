607 bool operator==(
608  const Variant<TFieldBase, TMembers, TOptions...>& field1,
609  const Variant<TFieldBase, TMembers, TOptions...>& field2)
610 {
611  if (&field1 == &field2) {
612  return true;
613  }
614 
615  if (field1.currentFieldValid() != field2.currentFieldValid()) {
616  return false;
