69 template <typename TUnsignedByteType, typename T>
70 typename std::decay<T>::type signExtCommon(T value, std::size_t size)
71 {
72  using ValueType = typename std::decay<T>::type;
73  static_assert(std::is_unsigned<ValueType>::value, "Type T must be unsigned");
74  static_assert(std::is_unsigned<TUnsignedByteType>::value,
75  "Type TUnsignedByteType must be unsigned");
76 
77  static const std::size_t BinDigits =
78  std::numeric_limits<TUnsignedByteType>::digits;
79  static_assert(BinDigits % 8 == 0, "Byte size assumption is not valid");
80 
81  ValueType mask =
82  (static_cast<ValueType>(1) << ((size * BinDigits) - 1));
83  if (value & mask) {
84  return value | (~((mask << 1) - 1));
85  }
86  return value;
87 }

315 {
316  using ValueType = typename std::decay<T>::type;
317  static_assert(std::is_integral<ValueType>::value, "T must be integral type");
318  static_assert(std::is_unsigned<ValueType>::value, "T type must be unsigned");

324  static_assert(BinDigits % 8 == 0, "Byte size assumption is not valid");
